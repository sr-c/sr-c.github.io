with open('rosalind_revc.txt') as string_dna:
    string_dna = string_dna.readline().strip()
    new_sting = []
    for i in string_dna:
        if i == 'A':
            new_sting.append('T')
        elif i == 'C':
            new_sting.append('G')
        elif i == 'T':
            new_sting.append('A')
        elif i == 'G':
            new_sting.append('C')

    #将列表倒序排列，revesre()函数进行原为改变，此时不能够将其赋值给另一个列表
    new_sting.reverse()
    #将列表转换为字符串
    rev_string = ''.join(new_sting)

print(rev_string)

try:
    fasta = {}
    with open('rosalind_gc.txt') as lines:
        for line in lines:
            line = line.strip()
            if line[0] == '>':
                name = line[1:]
            else:
                fasta[name] = fasta.get(name, '') + line

except IOError as err:
    print('File error: ' + str(err))

max_gc = 0
max_id = ''
#使用for循环取得最大的GC含量
for read_id in fasta.keys():
    rate_gc = (fasta[read_id].count('G') + fasta[read_id].count('C')) / len(fasta[read_id])
    if rate_gc > max_gc:
        max_gc = rate_gc
        max_id = read_id

print(max_id)
print(max_gc * 100)

上述程序要求输入的数据中，不含有N碱基，也不含有小写碱基。
